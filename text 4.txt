<script>
/*
  DEMO NOTES:
  - This is a purely client-side mock for prototyping only.
  - Demo credentials (embedded here only for demonstration):
      username: bobeda15
      password: Bobeda65$$1
      security Q: pet name
      security A: xena
    Do NOT use these on any production system.
*/

const DEMO = {
  username: "bobeda15",
  password: "Bobeda65$$1",
  securityQuestion: "pet name",
  securityAnswer: "xena",
  balance: 650789.79
};

const $ = id => document.getElementById(id);

const signinBtn = $('signin-btn');
const usernameInput = $('username');
const passwordInput = $('password');
const loginMessage = $('login-message');
const dashboard = $('dashboard');
const loginCard = $('login-card');
const balanceEl = $('balance-amount');
const displayUser = $('display-user');
const logoutBtn = $('logout');
const showRecovery = $('show-recovery');
const recoveryPanel = $('recovery-panel');
const recoveryCheck = $('recovery-check');
const recoveryCancel = $('recovery-cancel');
const recoveryMessage = $('recovery-message');
const recoveryAnswer = $('recovery-answer');

function formatCurrency(n){
  return n.toLocaleString('en-US', {style:'currency', currency:'USD'});
}

function showError(msg){
  loginMessage.textContent = msg;
  loginMessage.classList.remove('hidden');
}
function hideError(){
  loginMessage.classList.add('hidden');
}
signinBtn.addEventListener('click', () => {
  hideError();
  const u = usernameInput.value.trim();
  const p = passwordInput.value;
  if(!u || !p){ showError('Please enter username and password.'); return; }
  // Client-side demo auth check
  if(u === DEMO.username && p === DEMO.password){
    // Successful demo login
    loginCard.style.display = 'none';
    dashboard.style.display = 'block';
    balanceEl.textContent = formatCurrency(DEMO.balance);
    displayUser.textContent = DEMO.username;
    // Clear inputs for safety
    usernameInput.value = '';
    passwordInput.value = '';
    // Optionally show that this is a demo
    $('secret').classList.remove('hidden');
  } else {
    showError('Incorrect username or password (demo). Try recovery if needed.');
  }
});

// Recovery flow
showRecovery.addEventListener('click', () => {
  recoveryPanel.classList.remove('hidden');
  showRecovery.classList.add('hidden');
});
recoveryCancel.addEventListener('click', () => {
  recoveryPanel.classList.add('hidden');
  showRecovery.classList.remove('hidden');
  recoveryMessage.classList.add('hidden');
  recoveryAnswer.value = '';
});
recoveryCheck.addEventListener('click', () => {
  recoveryMessage.classList.add('hidden');
  const ans = recoveryAnswer.value.trim().toLowerCase();
  if(!ans){ recoveryMessage.textContent = 'Please enter an answer.'; recoveryMessage.classList.remove('hidden'); return; }
  if(ans === DEMO.securityAnswer.toLowerCase()){
    // In a real system you would verify identity and reset password via secure server.
    recoveryMessage.textContent = 'Security answer verified (demo). You may sign in with the demo password.';
    recoveryMessage.classList.remove('hidden');
    recoveryMessage.classList.remove('error');
    recoveryMessage.classList.add('success');
  } else {
    recoveryMessage.textContent = 'Incorrect answer (demo).';
    recoveryMessage.classList.remove('hidden');
    recoveryMessage.classList.remove('success');
    recoveryMessage.classList.add('error');
  }
});

// logout
logoutBtn.addEventListener('click', () => {
  dashboard.style.display = 'none';
  loginCard.style.display = 'block';
  recoveryPanel.classList.add('hidden');
  showRecovery.classList.remove('hidden');
  recoveryMessage.classList.add('hidden');
});
</script>
